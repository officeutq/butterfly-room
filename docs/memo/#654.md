```erb
<% content_for :title, "店舗管理者招待" %>

<div class="card mb-4" data-bs-theme="light">
  <div class="card-body">
    <p class="text-muted mb-3">
      招待URLを発行し、店舗管理者が承認した場合のみ、店舗に所属します。
    </p>

    <%= button_to "招待URLを発行",
          admin_store_admin_invitations_path,
          method: :post,
          class: "btn btn-primary" %>

    <div class="mt-3 text-muted small">
      ※発行後、招待URLをコピーして共有してください。
    </div>
  </div>
</div>

<h2 class="h5 mt-4">招待一覧</h2>

<% if @store_admin_invitations.blank? %>
  <div class="alert alert-warning" role="alert">
    発行済みの店舗管理者招待はありません。
  </div>
<% else %>
  <div class="referral-codes-list">
    <% @store_admin_invitations.each do |inv| %>
      <div class="card referral-code-card <%= "text-muted" unless inv.usable? %>">
        <div class="card-body">
          <div class="referral-code-row">
            <div class="referral-code-label">状態</div>
            <div class="referral-code-value">
              <div class="d-flex flex-wrap gap-2">
                <% if inv.usable? %>
                  <span class="badge text-bg-success">有効</span>
                <% elsif inv.used? %>
                  <span class="badge text-bg-secondary">使用済み</span>
                <% elsif inv.expired? %>
                  <span class="badge text-bg-danger">期限切れ</span>
                <% else %>
                  <span class="badge text-bg-secondary"><%= inv.status_label %></span>
                <% end %>
              </div>
            </div>
          </div>

          <div class="referral-code-row">
            <div class="referral-code-label">期限</div>
            <div class="referral-code-value">
              <%= l(inv.expires_at, format: :long) %>
            </div>
          </div>

          <div class="referral-code-row">
            <div class="referral-code-label">発行者</div>
            <div class="referral-code-value d-flex align-items-center gap-2">
              <%= user_avatar_badge(inv.invited_by_user, size: 28) %>
              <span><%= display_name_or_anonymous(inv.invited_by_user) %></span>
            </div>
          </div>

          <div class="referral-code-row">
            <div class="referral-code-label">承認者</div>
            <div class="referral-code-value d-flex align-items-center gap-2">
              <% if inv.accepted_by_user.present? %>
                <%= user_avatar_badge(inv.accepted_by_user, size: 28) %>
                <span><%= display_name_or_anonymous(inv.accepted_by_user) %></span>
              <% else %>
                -
              <% end %>
            </div>
          </div>

          <div class="referral-code-row">
            <div class="referral-code-label">招待URL</div>

            <div class="referral-code-value">
              <% if inv.issued_url.present? %>
                <div class="referral-code-url-group">
                  <code class="referral-code-url-inline"><%= inv.issued_url %></code>

                  <button
                    type="button"
                    class="btn btn-outline-secondary btn-sm referral-code-copy-btn"
                    data-controller="clipboard"
                    data-action="click->clipboard#copy"
                    data-clipboard-text="<%= inv.issued_url %>"
                    data-clipboard-success-message="招待URLをコピーしました"
                    data-clipboard-failure-message="招待URLのコピーに失敗しました"
                    aria-label="招待URLをコピー"
                    title="招待URLをコピー"
                  >
                    <i class="bi bi-copy" aria-hidden="true"></i>
                  </button>
                </div>
              <% else %>
                <span class="text-muted small">（発行時に控えてください）</span>
              <% end %>
            </div>
          </div>
        </div>
      </div>
    <% end %>
  </div>
<% end %>
```

* 発行者 / 承認者がアバター付きで出る
* 招待URLがコピーできる
* スマホ幅でも崩れない
