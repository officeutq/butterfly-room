```erb
<% primary_cast = booth.primary_cast_user %>

<div class="viewer-waiting-shell">
  <!-- hero -->
  <div class="booth-show-hero">
    <% if booth.thumbnail_image.attached? %>
      <%= image_tag booth.thumbnail_image, class: "booth-show-hero-image", alt: booth.name %>
    <% else %>
      <div class="booth-show-hero-placeholder">
        <span class="text-muted">no thumbnail</span>
      </div>
    <% end %>
  </div>

  <div class="container booth-show-container">
    <!-- header -->
    <div class="booth-show-header d-flex justify-content-between align-items-start gap-3">
      <div class="min-w-0">
        <h1 class="booth-show-title mb-0"><%= booth.name %></h1>
      </div>

      <div
        class="booth-show-favorite"
        data-viewer-page-target="boothFavoriteAnchor"
        aria-hidden="true"
      ></div>
    </div>

    <!-- info -->
    <div class="booth-show-info mt-4">
      <div class="booth-show-info-row">
        <div class="booth-show-info-label">キャスト</div>
        <div class="booth-show-info-value">
          <% if primary_cast.present? %>
            <%= link_to user_path(primary_cast), class: "text-decoration-none d-inline-flex align-items-center gap-2" do %>
              <%= user_avatar_badge(primary_cast, size: 28) %>
              <span><%= display_name_or_anonymous(primary_cast) %></span>
            <% end %>
          <% else %>
            <span class="text-muted">（未設定）</span>
          <% end %>
        </div>
      </div>

      <div class="booth-show-info-row">
        <div class="booth-show-info-label">店舗</div>
        <div class="booth-show-info-value">
          <%= link_to booth.store.name,
                store_path(booth.store),
                class: "text-decoration-none fw-semibold" %>

          <span
            data-viewer-page-target="storeFavoriteAnchor"
            aria-hidden="true"
          ></span>
        </div>
      </div>
    </div>

    <!-- description -->
    <div class="booth-show-description mt-4">
      <%= simple_format(booth.description.presence || "-") %>
    </div>
  </div>
</div>
```
