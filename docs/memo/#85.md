## 3) 実装

### 3.1 `Authorization::BoothPolicy` を拡張

`app/services/authorization/booth_policy.rb`

```ruby
module Authorization
  class BoothPolicy < ApplicationPolicy
    def show?
      true
    end

    def cast_operate?
      user.cast? || user.system_admin?
    end

    def admin_operate?
      user.store_admin? || user.system_admin?
    end

    # --- 追加: Booth編集権限 ---
    def edit?
      update?
    end

    def update?
      return false if user.blank?

      return true if user.system_admin?
      return false if user.customer?

      if user.store_admin?
        # store_admin：自分の store の booth のみ編集可
        return user.store_memberships.admin_only.exists?(store_id: record.store_id)
      end

      if user.cast?
        # cast：所属する booth のみ編集可
        return record.booth_casts.exists?(cast_user_id: user.id)
      end

      false
    end
  end
end
```

これで仕様の4ロールが綺麗に入ります。

### 3.2 Adminの update を 403 ガード

`app/controllers/admin/booths_controller.rb`

`update` がまだ stub でも、まずは guard を入れるのが #85 の目的に合います。

```ruby
module Admin
  class BoothsController < Admin::BaseController
    before_action :require_current_store!, only: %i[watch]
    before_action :set_booth, only: %i[show update watch]
    before_action :authorize_update!, only: %i[update]

    def index
      render plain: "admin booths index (stub)"
    end

    def show
      render plain: "admin booths show (stub)"
    end

    def update
      head :not_implemented
    end

    def watch
      policy = Authorization::BoothPolicy.new(current_user, @booth)
      head :forbidden and return unless policy.admin_operate?

      stream_session = StreamSession.find_by(id: @booth.current_stream_session_id)

      if stream_session.blank? || !stream_session.live?
        redirect_to admin_booth_path(@booth), alert: "配信中ではありません"
        return
      end

      redirect_to booth_path(@booth)
    end

    private

    def set_booth
      @booth = Booth.find(params[:id])
    end

    def authorize_update!
      policy = Authorization::BoothPolicy.new(current_user, @booth)
      head :forbidden unless policy.update?
    end
  end
end
```

## 4) テスト（#85のDone条件を満たす：ここに集約）

### 4.1 どのURLを叩くか

* customer/cast 系 → `cast` 名前空間の `edit` を叩く（編集機能の入口）
* store_admin/system_admin 系 → `admin` 名前空間の `update` を叩く（現状 routes に update がある）

### 4.2 テスト雛形（fixturesが無い前提で自前作成）

あなたは fixtures :all を使ってるけど、今ここで fixtures の中身が分からないので、**テスト内で create! して完結**させるのが安全です。

`test/integration/booth_edit_authorization_test.rb`（新規）

```ruby
# frozen_string_literal: true

require "test_helper"

class BoothEditAuthorizationTest < ActionDispatch::IntegrationTest
  setup do
    @store1 = Store.create!(name: "store1")
    @store2 = Store.create!(name: "store2")

    @booth1 = Booth.create!(store: @store1, name: "booth1", status: :offline)
    @booth2 = Booth.create!(store: @store2, name: "booth2", status: :offline)

    @customer     = User.create!(email: "customer@example.com", password: "password", role: :customer)
    @cast         = User.create!(email: "cast@example.com", password: "password", role: :cast)
    @store_admin  = User.create!(email: "admin@example.com", password: "password", role: :store_admin)
    @system_admin = User.create!(email: "sys@example.com", password: "password", role: :system_admin)

    # cast は booth1 に所属
    BoothCast.create!(booth: @booth1, cast_user: @cast)

    # store_admin は store1 の admin
    StoreMembership.create!(store: @store1, user: @store_admin, membership_role: :admin)
  end

  test "customer cannot edit/update (403)" do
    sign_in @customer, scope: :user

    get edit_cast_booth_path(@booth1)
    assert_response :forbidden

    patch cast_booth_path(@booth1), params: { booth: { name: "x" } }
    assert_response :forbidden

    patch admin_booth_path(@booth1), params: { booth: { name: "x" } }
    assert_response :forbidden
  end

  test "cast can edit only own booth" do
    sign_in @cast, scope: :user

    get edit_cast_booth_path(@booth1)
    assert_response :success

    get edit_cast_booth_path(@booth2)
    assert_response :forbidden
  end

  test "store_admin can update only own store booth" do
    sign_in @store_admin, scope: :user

    patch admin_booth_path(@booth1), params: { booth: { name: "x" } }
    assert_response :not_implemented # update stubならこれでOK（403にならないのが重要）

    patch admin_booth_path(@booth2), params: { booth: { name: "x" } }
    assert_response :forbidden
  end

  test "system_admin can update any booth" do
    sign_in @system_admin, scope: :user

    patch admin_booth_path(@booth1), params: { booth: { name: "x" } }
    assert_response :not_implemented

    patch admin_booth_path(@booth2), params: { booth: { name: "x" } }
    assert_response :not_implemented
  end
end
```

> `Store` の必須カラムが `name` 以外にあるなら、そこだけあなたのschemaに合わせて追記してください（ここだけは手元のschema次第）。

## 5) 落とし穴（重要）

* `User` に `booth_casts` 関連が無いので、**user側から辿らない**（今回のPolicyは record側から辿るのでOK）
* store_admin 判定は membership_role を必ず効かせる（`admin_only` を使うのが正）
* テストは「403になるべき」だけじゃなく「403にならないべき」も見る
  → 今回は `:not_implemented` を“通過できた証拠”として使うのがコツ
