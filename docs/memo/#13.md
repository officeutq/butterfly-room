## 1. Devise導入

### 1) Gem 追加と初期設定

* Gem 追加  
   ```bash
   docker compose exec app bundle add devise
   docker compose exec app rails g devise:install
   ```

* `.rubocop.yml` で`config/initializers/devise.rb`を除外
   ```yaml
   AllCops:
     Exclude:
       - 'config/initializers/devise.rb'
   ```

* 生成された `config/initializers/devise.rb` に **Turbo 互換** を追加（重要）：

   ```ruby
   # Turbo (Hotwire) と共存
   config.navigational_formats = ['*/*', :html, :turbo_stream]
   ```

### 2) Usersを生成

```bash
docker compose run --rm -e DISABLE_SPRING=1 app bin/rails g migration CreateUsers
```
`db/migrate/xxxxxxxxxxxxxx_create_users.rb`
```rb
class CreateUsers < ActiveRecord::Migration[8.1]
```

### 3.2 store_memberships

* membership_role: enum（cast/admin）
* unique(store_id, user_id, membership_role)
* index(store_id, membership_role), index(user_id)

```rb
class CreateStoreMemberships < ActiveRecord::Migration[8.1]
  def change
    create_table :store_memberships do |t|
      t.references :store, null: false, foreign_key: true
      t.references :user,  null: false, foreign_key: true
      t.integer :membership_role, null: false

      t.timestamps
    end

    add_index :store_memberships, %i[store_id user_id membership_role], unique: true
    add_index :store_memberships, %i[store_id membership_role]
  end
end
```

---

### 3.3 booths

* status: offline/live/away
* current_stream_session_id: nullable（FKは後で）

```rb
class CreateBooths < ActiveRecord::Migration[8.1]
  def change
    create_table :booths do |t|
      t.references :store, null: false, foreign_key: true
      t.string  :name, null: false
      t.integer :status, null: false
      t.bigint  :current_stream_session_id # FKは後で付ける

      t.timestamps
    end

    add_index :booths, %i[store_id status]
    add_index :booths, :current_stream_session_id
  end
end
```

### 3.5 stream_sessions

* status: live/ended
* started_at 必須、ended_at nullable
* store_id 冗長に持つ（設計通り）
* started_by_cast_user_id

```rb
class CreateStreamSessions < ActiveRecord::Migration[8.1]
  def change
    create_table :stream_sessions do |t|
      t.references :booth, null: false, foreign_key: true
      t.references :store, null: false, foreign_key: true
      t.integer :status, null: false
      t.datetime :started_at, null: false
      t.datetime :ended_at
      t.references :started_by_cast_user, null: false, foreign_key: { to_table: :users }

      t.timestamps
    end

    add_index :stream_sessions, %i[booth_id started_at]
    add_index :stream_sessions, %i[store_id started_at]
    add_index :stream_sessions, :ended_at
  end
end
```

### 3.8 wallets（unique + 非負制約）

```rb
class CreateWallets < ActiveRecord::Migration[8.1]

    add_check_constraint :wallets, "reserved_points >= 0",  name: "wallets_reserved_points_non_negative"
  end
end
```

### 3.11 drink_orders（最重要：FIFO index）

**必須**：`(stream_session_id, status, created_at, id)`

```rb
class CreateDrinkOrders < ActiveRecord::Migration[8.1]
  def change
    create_table :drink_orders do |t|
      t.references :store, null: false, foreign_key: true
      t.references :booth, null: false, foreign_key: true
      t.references :stream_session, null: false, foreign_key: true
      t.references :customer_user, null: false, foreign_key: { to_table: :users }
      t.references :drink_item, null: false, foreign_key: true

      t.integer :status, null: false
      t.datetime :consumed_at
      t.datetime :refunded_at

      t.timestamps
    end

    add_index :drink_orders, %i[stream_session_id status created_at id], name: "idx_drink_orders_fifo"
    add_index :drink_orders, %i[customer_user_id created_at], order: { created_at: :desc }
    add_index :drink_orders, %i[store_id status created_at]
    add_index :drink_orders, %i[store_id consumed_at]
  end
end
```
